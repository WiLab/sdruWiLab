1classdef OFDMDemodulator < OFDMBase
  2%OFDMDemodulator Apply OFDM demodulation to the input signal.
  3%   H = OFDMDemodulator returns a System object, H, that can be used to
  4%   transform OFDM symbol from time domain to frequency domain and separate
  5%   data from guard bands, pilots, or preambles when applicable.
  6%
  7%   H = OFDMModulator('Name', Value, ...) returns an OFDM Demodulator
  8%   System object, H, with each specified property name set to the 
  9%   specified value. You can specify name-value pair arguments in any order 
 10%   as (Name 1, Value 1, Name 2, Value 2, ....).
 11%
 12%   H = OFDMDemodulator(hMod) returns an OFDM Demodulator System
 13%   object, H, based on an OFDM Modulator hMod.
 14%
 15%   Step method syntax:
 16%
 17%   Y = step(H, X) computes the OFDM demodulation, Y, of input X. Depending
 18%   on the property settings, there can be one to three outputs:
 19%   -- Y = step(H, X), Y is the transmitted data symbol
 20%   -- [Y, Y2] = step(H,X), Y2 is either the preamble or the pilot sequence
 21%      depending on the properties. Y is the same as above
 22%   -- [Y, Y2, Y3] = step(H,X), Y2 is the preamble sequence, Y3 is the pilot
 23%      sequency and Y is the same as above
 24%
 25%   OFDMDemodulator methods: 
 26%
 27%   ShowResourceMapping  - Display the grid layout with information about the location
 28%   of data, preamble, pilots, nulls, and guardbands
 29%      
 30%   OFDMDemodulator properties:
 31%
 32%   FFTLength                  - IFFT length
 33%   GuardBands                 - Guard bands
 34%   CyclicPrefixLength         - Cyclic prefix length
 35%   NullsType                  - Type of nulls inserted
 36%   NullCarrierIndices         - Null locations on the grid
 37%   PilotOutputPort             - Option to insert pilots
 38%   PilotCarrierIndices        - Pilot locations on the grid
 39%   SeparatePreamble           - Option to enable preamble input
 40%   NumPreambleSymbols         - Number of preamble symbols
 41%   PreambleGuardBands         - Preamble guard bands
 42%   PreambleCyclicPrefixLength - Preamble cyclic prefix length
 43%   NumInputSymbols            - Number of input OFDM symbols
 44%   NumReceiveAntennas         - Number of Receive antennas
 45
 46%   Copyright 2013 The MathWorks, Inc.
 47
 48%#codegen
 49%#ok<*EMCA>
 50
 51properties (Nontunable)
 52    %NumReceiveAntennas Number of receive antennas
 53    % Specify the number of tranmit antennas to be used. This property
 54    % can be set to a positive integer scaler. The default value of this property is 1.
 55    NumReceiveAntennas = 1;
 56end
 57
 58properties (Nontunable, Logical)
 59    %RemoveDCCarrier Remove DC subcarrier from output
 60    % Specify the nulls type to be inserted: 'None', 'DC-only',
 61    % 'Custom'. The default setting of this property is 'DC-only'
 62    RemoveDCCarrier = false;
 63    %PilotOutputPort Pilot output port
 64    % Setting this property to true, the System object will expect a
 65    % separate pilot input. The default value of this property is false.
 66    PilotOutputPort = false;
 67end

 70  function object = OFDMDemodulator(varargin)
 71    if nargin == 1 && isa(varargin{1}, 'OFDMModulator')
 72        modObj = varargin{1};        
 73        object.FFTLength            = modObj.FFTLength;
 74        object.NumGuardBandCarriers = modObj.NumGuardBandCarriers;     
 75        object.CyclicPrefixLength   = modObj.CyclicPrefixLength;
 76        object.RemoveDCCarrier      = modObj.InsertDCNull;
 77        object.NumSymbols           = modObj.NumSymbols;
 78        object.PilotOutputPort      = modObj.PilotInputPort;
 79        if modObj.PilotInputPort
 80            object.PilotCarrierIndices = modObj.PilotCarrierIndices;
 81        end        
 82    else
 83        setProperties(object, nargin, varargin{:});
 84    end
 85  end

113  function setupImpl(obj, varargin)        
114    FFTLen = obj.FFTLength;
115    numSym = obj.NumSymbols;
116    numRx  = obj.NumReceiveAntennas;
117    
118    % Fill fill grid for demodulator 
119    fullGrid = ones(FFTLen, numSym, numRx);
120    fullGrid([1:obj.NumGuardBandCarriers(1), ...
121        FFTLen-obj.NumGuardBandCarriers(2)+1:FFTLen], :, :) = 0; % Guard bands = 0
122    if obj.RemoveDCCarrier
123        fullGrid(FFTLen/2+1, :, :) = 0; % DC null = 0
124    end
125    if obj.PilotOutputPort % Pilot = 2
126        % Calculate pilot 3D indices at Rx
127        pilotIdx = obj.PilotCarrierIndices;    
128        if ismatrix(pilotIdx)
129            pilotRx3DIndices = repmat(pilotIdx, [1 numSym/size(pilotIdx,2) numRx]);
130        else
131            pilotRx3DIndices = repmat(reshape(permute(pilotIdx, [1 3 2]), ...
132                size(pilotIdx, 1)*size(pilotIdx, 3), size(pilotIdx, 2)), ...
133                [1 numSym/size(pilotIdx,2), numRx]);
134        end
135        % Calculate pilot linear indices at Rx
136        obj.pPilotLinearIndices = convert3DIndexToLinearIndex(obj, pilotRx3DIndices);
137        fullGrid(obj.pPilotLinearIndices) = 2;
138    end    
139    
140    % Calculate data linear indices at Rx
141    obj.pDataLinearIndices  = find(fullGrid == 1);
142  end

144  function varargout = stepImpl(obj, x)
145    FFTLen = obj.FFTLength;
146    CPLen  = obj.CyclicPrefixLength;
147    numSym = obj.NumSymbols;
148    numRx  = obj.NumReceiveAntennas;        
149      
150    % Remove Cyclic prefix
151    if isscalar(CPLen)
152        inputIn3D = reshape(x(1:end, :), [(FFTLen + CPLen), numSym, numRx]);
153        preFFT = inputIn3D(CPLen+1:end, :, :);
154    else
155        preFFT = coder.nullcopy(complex(zeros(FFTLen, numSym, numRx)));
156        for symIdx = 1:numSym
157            preFFT(:, symIdx, :) = x(FFTLen*(symIdx-1)+sum(CPLen(1:symIdx))+(1:FFTLen), :);
158        end
159    end
160
161    % FFT: time domain to frequency domain
162    postFFT = fft(preFFT, [], 1);
163
164    % FFT shift
165    if isreal(postFFT)
166        postShift = complex(fftshift(postFFT, 1), 0);
167    else
168        postShift = fftshift(postFFT,1);
169    end
170    
171    varargout{1} = reshape(postShift(obj.pDataLinearIndices), ...
172        [length(obj.pDataLinearIndices)/numSym/numRx, numSym, numRx]);
173    
174    if obj.PilotOutputPort
175        if ismatrix(obj.PilotCarrierIndices)
176            varargout{2} = reshape(postShift(obj.pPilotLinearIndices), ...
177                [length(obj.pPilotLinearIndices)/numSym/numRx, numSym, numRx]);
178        else
179            numTx = size(obj.PilotCarrierIndices, 3);
180            varargout{2} = permute(reshape(postShift(obj.pPilotLinearIndices), ...
181                [length(obj.pPilotLinearIndices)/numTx/numSym/numRx, numTx, numSym, numRx]), ...
182                [1 3 2 4]);
183        end
184    end
185  end

204  function s = infoImpl(obj)          
205    if ~isLocked(obj)        
206        validateProperties(obj);
207    end
208    
209    numSym = obj.NumSymbols;
210    numRx  = obj.NumReceiveAntennas;
211    
212    s.InputSize = [getTimeDomainFrameLength(obj) numRx];
213
214    if numRx == 1
215        s.DataOutputSize = [getNumDataCarriers(obj, obj.RemoveDCCarrier, obj.PilotOutputPort) numSym];
216    else
217        s.DataOutputSize = [getNumDataCarriers(obj, obj.RemoveDCCarrier, obj.PilotOutputPort) numSym numRx];
218    end
219    
220    if obj.PilotOutputPort
221        if ismatrix(obj.PilotCarrierIndices)
222            s.PilotOutputSize = [size(obj.PilotCarrierIndices, 1) numSym];
223        else
224            s.PilotOutputSize = [size(obj.PilotCarrierIndices, 1) numSym size(obj.PilotCarrierIndices, 3)];
225        end
226        
227        if numRx > 1
228            s.PilotOutputSize = [s.PilotOutputSize numRx];
229        end
230    end
231  end
