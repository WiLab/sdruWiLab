1function reportSDRuStatus(errStatus, errMsg, ipAddress, reportMethod)
 2%reportSDRuStatus SDRu status reporter
 3%   reportSDRuStatus(STATUS,MSG,IP,METHOD) reports the SDRu status based on
 4%   the STATUS input. STATUS is a UsrpErrorCapiEnumT type. IP is the IP
 5%   address of the USRP(R) radio. METHOD is a UsrpReportMethodEnumT type and
 6%   can be Warning, Error , or None.
 7%
 8%   USRP(R) is a trademark of National Instruments Corp.
 9
10%   Copyright 2012-2013 The MathWorks, Inc.
11
12%#codegen
13
14  coder.internal.errorIf(...
15    ~isa(errStatus, 'UsrpErrorCapiEnumT') || ~isscalar(errStatus), ...
16    'sdru:reportSDRuStatus:InvalidErrorStatus')
17
18  coder.internal.errorIf(~ischar(errMsg), ...
19    'sdru:reportSDRuStatus:InvalidErrorMessage', errMsg);
20
21  coder.internal.errorIf(~ischar(ipAddress), ...
22    'sdru:reportSDRuStatus:InvalidIPAddress');
23  coder.internal.errorIf(~isrow(ipAddress), ...
24    'sdru:reportSDRuStatus:InvalidIPAddress');
25
26  coder.internal.errorIf(...
27    ~isa(reportMethod, 'UsrpReportMethodEnumT') || ~isscalar(reportMethod), ...
28    'sdru:reportSDRuStatus:InvalidReportMethod')
29
30  switch errStatus
31    case UsrpErrorCapiEnumT.UsrpDriverNotCompatible
32      msgID = 'sdru:reportSDRuStatus:NotCompatible';
33      switch reportMethod
34        case UsrpReportMethodEnumT.Warning
35          coder.internal.warning(msgID, ipAddress, getSDRuDriverVersion);
36        case UsrpReportMethodEnumT.Error
37          if isempty(coder.target)
38            coder.internal.errorIf(...
39              errStatus == UsrpErrorCapiEnumT.UsrpDriverNotCompatible, ...
40              msgID, ipAddress, getSDRuDriverVersion);
41          else
42            coder.internal.errorIf(...
43              errStatus == UsrpErrorCapiEnumT.UsrpDriverNotCompatible, ...
44              msgID, ipAddress, 'returned by ''getSDRuDriverVersion'' function');
45          end
46      end
47    case UsrpErrorCapiEnumT.UsrpDriverNotResponding
48      msgID = 'sdru:reportSDRuStatus:NotResponding';
49      switch reportMethod
50        case UsrpReportMethodEnumT.Warning
51          coder.internal.warning(msgID, ipAddress);
52        case UsrpReportMethodEnumT.Error
53          coder.internal.errorIf(...
54            errStatus == UsrpErrorCapiEnumT.UsrpDriverNotResponding, ...
55            msgID, ipAddress);
56      end
57    case UsrpErrorCapiEnumT.UsrpDriverRxBusy
58      msgID = 'sdru:reportSDRuStatus:RxBusy';
59      switch reportMethod
60        case UsrpReportMethodEnumT.Warning
61          coder.internal.warning(msgID, ipAddress);
62        case UsrpReportMethodEnumT.Error
63          coder.internal.errorIf(...
64            errStatus == UsrpErrorCapiEnumT.UsrpDriverRxBusy, ...
65            msgID, ipAddress);
66      end
67    case UsrpErrorCapiEnumT.UsrpDriverTxBusy
68      msgID = 'sdru:reportSDRuStatus:TxBusy';
69      switch reportMethod
70        case UsrpReportMethodEnumT.Warning
71          coder.internal.warning(msgID, ipAddress);
72        case UsrpReportMethodEnumT.Error
73          coder.internal.errorIf(...
74            errStatus == UsrpErrorCapiEnumT.UsrpDriverTxBusy, ...
75            msgID, ipAddress);
76      end
77    case UsrpErrorCapiEnumT.UsrpDriverBusy
78      msgID = 'sdru:reportSDRuStatus:Busy';
79      switch reportMethod
80        case UsrpReportMethodEnumT.Warning
81          coder.internal.warning(msgID, ipAddress);
82        case UsrpReportMethodEnumT.Error
83          coder.internal.errorIf(...
84            errStatus == UsrpErrorCapiEnumT.UsrpDriverBusy, ...
85            msgID, ipAddress);
86      end
87    case UsrpErrorCapiEnumT.UsrpDriverError
88      msgID = 'sdru:reportSDRuStatus:UnknownStatus';
89      switch reportMethod
90        case UsrpReportMethodEnumT.Warning
91          coder.internal.warning(msgID, errMsg)
92        case UsrpReportMethodEnumT.Error
93          coder.internal.errorIf(...
94            errStatus == UsrpErrorCapiEnumT.UsrpDriverError, msgID, errMsg)
95      end
96  end
97end
