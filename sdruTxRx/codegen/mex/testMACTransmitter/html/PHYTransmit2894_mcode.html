1% Send Messages
 2function PHYTransmit(...
 3    ObjSDRuTransmitter,...
 4    ObjSDRuReceiver,...
 5    inputPayloadMessage,...
 6    samplingFreq,...
 7    originNodeID,...
 8    destNodeID...
 9    )
10
11DebugFlag = 0;
12
13%% Create message bits
14% The output needs to be duplicated to long vectors to help prevent
15% Underflow for the USRP
16[~,~, dataToTx, ~ ] = generateOFDMSignal_TX2(inputPayloadMessage,samplingFreq, originNodeID, destNodeID);% 30 Dupe frames created (NOTE! author shouldcreate shorter simpler function)
17
18%% Run transmitter
19if DebugFlag
20    fprintf('PHY| Transmitting... pew! pew!\n');
21    fprintf('PHY| TX Message: %s\n',inputPayloadMessage);
22end
23
24numFramesToTx = 10;%increasing value will help receiver, 10
25% This should be longer to help transmit over periods when the RX is
26% cleaning its buffer
27
28for framesTransmitted = 1:numFramesToTx
29    step(ObjSDRuTransmitter, dataToTx);
30    
31    %if mod(framesTransmitted,60) == 0
32    step(ObjSDRuReceiver);% Call used to prevent Overflow.  Essentially will clean up receive buffer, will be filled with crosstalk frames
33    %end
34end
35
36%obj.pSDRuTransmitter.reset;%stop transmitting?
37
38end
