1% This function is written according to (1) Figure 3 and Figure 5 of the
 2% reference: E. L. Da Costa, "Detection and Identification of
 3% Cyclostationary Signals". MS Thesis. 1996. (2) Section 3.2 of the
 4% reference: Eric April, "On the Implementation of the Strip Spectral
 5% Correlation Algorithm for Cyclic Spectrum Estimation". February, 1994.
 6
 7%   Copyright 2012 The MathWorks, Inc.
 8
 9function [Sx,alphao,fo] = commP25sscaGen(input,fs,df,dalpha)
10% fs is sampling frequency, df is spectral resolution, dalpha is cyclic
11% resolution
12narginchk(4,4);              % check the number of input variables
13
14% Definition of Parameters
15N = fs/dalpha;               % observation time T in discrete domain
16Np = fs/df;                  % window time Tw in discrete domain, Eq (18)
17L = Np/4;                    % number of overlap samples
18P = fs/(L*dalpha);           % Eq (19), we can also get N = P*L
19alphao = (-1:1/N:1)*fs;      % cyclic frequency (alpha)
20fo = (-0.5:1/Np:0.5)*fs;     % spectral frequency (f)
21
22% Input vector processing, the length of input vector will be made to N
23if length(input)<N
24    input(N) = 0;           % pad zeros
25else
26    input = input(1:N);     % truncate input to length N
27end
28
29% Implement Figure 3 in (1): Computation of the Complex Demodulates
30% Construct input matrix in Figure 3, call it x
31NN = (P-1)*L+Np;            % number of elements in x matrix
32xx = input;
33if length(xx)<NN
34    xx(NN) = 0;             % pad zeros
35else    
36    xx = xx(1:NN);          % truncate input to length NN
37end
38x = complex(zeros(Np,P));            % x matrix initialization
39for k = 0:P-1
40    x(:,k+1) = xx(k*L+1:k*L+Np);
41end
42
43% Windowing
44a = hamming(Np);           % Construct N'-point Hamming window
45XWin = diag(a)*x;          % Output of N'-point Hamming window
46
47% Centered N'-point FFT (zero frequency in the 0 bin)
48XFFT1 = fftshift(fft(XWin)); 
49X = [XFFT1(:,P/2+1:P) XFFT1(:,1:P/2)];
50
51% Downshift in frequency
52E = complex(zeros(Np,P));
53for idxl = -Np/2:Np/2-1    % length of l is Np
54    for k = 0:P-1          % length of k is P
55        E(idxl+Np/2+1,k+1) = exp(-1i*2*pi*k*L*idxl/Np);
56    end
57end
58XT = X.*E;                 % Output of Figure 3
59
60% Implement Figure 5 in (1): Implementation of the SSCA
61% Replication, repeat the k-th column of XT (XT(:,k)) L times
62XRep = complex(zeros(Np,P*L));
63for k = 1:P
64    XRep(:,(k-1)*L+1:k*L) = XT(:,k)*ones(1,L);
65end
66
67% Multiplication
68% Convert input vector into a matrix, the vector first becomes a row
69% vector, and repeat Np times. xc is (Np,N), the same size as XR
70inputRep = ones(Np,1)*input';
71XMul = conj((XRep.*inputRep)');
72
73% Centered N-point FFT (zero frequency in the 0 bin)
74XFFT2 = fftshift(fft(XMul));
75XFFT2 = [XFFT2(:,Np/2+1:Np) XFFT2(:,1:Np/2)]; % Output of Figure 5
76
77M = abs(XFFT2);
78
79% Convert M to Sx
80Sx = zeros(Np+1,2*N+1);
81for k1 = 1:N
82    k2 = 1:Np;
83    alpha = (k1-1)/N + (k2-1)/Np - 1;
84    f = ((k2-1)/Np - (k1-1)/N) / 2;
85    k = 1 + Np*(f+0.5);
86    n = 1 + N*(alpha+1);
87    for k2 = 1:Np
88        Sx(round(k(k2)),round(n(k2))) = M(k1,k2);
89    end
90end
