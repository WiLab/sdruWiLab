1function [Response, previousMessage] = MACLayerReceiver(...
 2    ObjAGC,...           %Objects
 3    ObjSDRuReceiver,...
 4    ObjSDRuTransmitter,...
 5    ObjDetect,...
 6    ObjPreambleDemod,...
 7    ObjDataDemod,...
 8    estimate,...         %Structs
 9    tx,...
10    timeoutDuration,...  %Values/Vectors
11    messageBits,...
12    lookingForACK,...
13    previousMessage...
14    )
15
16% This function is called when the node is just listening to the spectrum
17% waiting for a message to be transmitted to them
18
19%% Listen to the spectrum
20% previousMessage will be updated for next run
21[Response, previousMessage, originNodeID, destNodeID] = DLLayer(...
22    ObjAGC,...           %Objects
23    ObjSDRuReceiver,...
24    ObjDetect,...
25    ObjPreambleDemod,...
26    ObjDataDemod,...
27    estimate,...         %Structs
28    tx,...
29    timeoutDuration,...  %Values/Vectors
30    messageBits,...
31    previousMessage...
32    );
33
34
35
36%% Possible response messages
37% 1.) Timeout
38% 2.) Some message
39if ~strcmp(Response,'Timeout') &&  ~strcmp(Response,'Duplicate')
40    %fprintf('###########################################\n');
41    fprintf('MAC| Got message: %s\n',Response);
42    fprintf('MAC| From Node: %d\n',int16(originNodeID));
43    
44    if ~lookingForACK
45        fprintf('MAC| Transmitting ACK\n');
46        % Send ACK
47        fprintf('Transmitting to node: %d, with offset: %f\n',int16(originNodeID),tx.offsetTable(originNodeID));
48        ObjSDRuTransmitter.CenterFrequency = tx.CenterFrequency + tx.offsetTable(originNodeID);% Adjust offset for node       
49        PHYTransmit(...
50            ObjSDRuTransmitter,...
51            ObjSDRuReceiver,...
52            'ACK',...
53            tx.samplingFreq,...
54            originNodeID...
55            );
56        
57    end
58elseif strcmp(Response,'Duplicate')
59    
60    fprintf('MAC| Transmitting ACK for dupe\n');
61    %% Send ACK
62    fprintf('Transmitting to node: %d, with offset: %f\n',int16(originNodeID),tx.offsetTable(originNodeID));
63    ObjSDRuTransmitter.CenterFrequency = tx.CenterFrequency + tx.offsetTable(originNodeID); % Adjust offset for node
64    PHYTransmit(...
65        ObjSDRuTransmitter,...
66        ObjSDRuReceiver,...
67        'ACK',...
68        tx.samplingFreq,...
69        originNodeID...
70        );
71end
72
73
74end
