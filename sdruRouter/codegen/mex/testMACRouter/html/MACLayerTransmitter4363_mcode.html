1function [previousMessage, msgStatus] = MACLayerTransmitter(...
 2    ObjAGC,...           %Objects
 3    ObjSDRuReceiver,...
 4    ObjSDRuTransmitter,...
 5    ObjDetect,...
 6    ObjPreambleDemod,...
 7    ObjDataDemod,...
 8    estimate,...         %Structs
 9    tx,...
10    timeoutDuration,...  %Values/Vectors
11    messageBits,...
12    message,...
13    previousMessage,...
14    recipient...
15    )
16
17%% This function is called when the node wants to transmit something
18
19% % Sense spectrum and wait until it is unoccupied
20% for tries = 1:4 % try only so many times
21%     occupied = PHY.Sense;
22%     if occupied
23%         fprintf('MAC| Spectrum occupied, listening...\n');
24%         %Recover signal and/or wait
25%         lookingForACK = false;
26%         MACLayerReceiver(PHY,lookingForACK);
27%     else% Yay we can transmit now
28%         break;
29%     end    
30%     if tries >=4
31%         fprintf('MAC| Spectrum Busy, try again later\n');
32%         return;
33%     end
34% end
35
36maxRetries = 4;
37
38msgStatus = false;
39
40% Adjust offset for node
41fprintf('Transmitting to node: %d, with offset: %f\n',int16(recipient),tx.offsetTable(recipient));
42ObjSDRuTransmitter.CenterFrequency = tx.CenterFrequency + tx.offsetTable(recipient);
43
44
45%% Spectrum clear, send message
46for tries = 1:maxRetries
47    % Send message
48    PHYTransmit(...
49        ObjSDRuTransmitter,...
50        ObjSDRuReceiver,...
51        message,...
52        tx.samplingFreq,...
53        tx.nodeNum,...%originator
54        recipient...%destination
55        );
56    % Listen for acknowledgement
57    %fprintf('###########################################\n');
58    fprintf('MAC| Transmission finished, waiting for ACK\n');
59    % Call Receiver
60    lookingForACK = true;
61    [Response, previousMessage] = MACLayerReceiver(...
62        ObjAGC,...           %Objects
63        ObjSDRuReceiver,...
64        ObjSDRuTransmitter,...
65        ObjDetect,...
66        ObjPreambleDemod,...
67        ObjDataDemod,...
68        estimate,...         %Structs
69        tx,...
70        timeoutDuration,...  %Values/Vectors
71        messageBits,...
72        lookingForACK,...
73        previousMessage...
74        );
75
76    if strcmp(Response,'ACK')
77        fprintf('MAC| Got ACK\n');
78        msgStatus = true;
79        break
80    else
81        fprintf('MAC| Retransmitting message\n');
82    end
83    if tries >= 4
84        fprintf('###########################################\n');
85        fprintf('MAC| No ACK received :(\n');
86        fprintf('###########################################\n');
87        return;
88    end 
89end
