1% This function is written according to (1) Figure 3 and Figure 5 of the
 2% reference: E. L. Da Costa, "Detection and Identification of
 3% Cyclostationary Signals". MS Thesis. 1996. (2) Section 3.2 of the
 4% reference: Eric April, "On the Implementation of the Strip Spectral
 5% Correlation Algorithm for Cyclic Spectrum Estimation". February, 1994.
 6
 7%   Copyright 2012 The MathWorks, Inc.
 8
 9function [Sx,alphao,fo] = commP25sscaGen(input,fs,df,dalpha)
10% fs is sampling frequency, df is spectral resolution, dalpha is cyclic
11% resolution
12%narginchk(4,4);              % check the number of input variables
13
14% Definition of Parameters
15N = fs/dalpha;               % observation time T in discrete domain
16Np = fs/df;                  % window time Tw in discrete domain, Eq (18)
17L = Np/4;                    % number of overlap samples
18P = fs/(L*dalpha);           % Eq (19), we can also get N = P*L
19alphao = (-1:1/N:1)*fs;      % cyclic frequency (alpha)
20fo = (-0.5:1/Np:0.5)*fs;     % spectral frequency (f)
21
22% Input vector processing, the length of input vector will be made to N
23if length(input)<N
24    input(N) = 0;           % pad zeros
25else
26    input = input(1:N);     % truncate input to length N
27end
28
29% Implement Figure 3 in (1): Computation of the Complex Demodulates
30% Construct input matrix in Figure 3, call it x
31NN = (P-1)*L+Np;            % number of elements in x matrix
32%xx = input;
33if length(input)<NN
34    %xx(NN) = 0;             % pad zeros
35    xx = [input;zeros(NN-length(input),1)];
36else    
37    xx = input(1:NN);          % truncate input to length NN
38end
39x = complex(zeros(Np,P));            % x matrix initialization
40for k = 0:P-1
41    x(:,k+1) = xx(k*L+1:k*L+Np);
42end
43
44% Windowing
45a = hamming(Np);           % Construct N'-point Hamming window
46XWin = diag(a)*x;          % Output of N'-point Hamming window
47
48% Centered N'-point FFT (zero frequency in the 0 bin)
49XFFT1 = fftshift(fft(XWin)); 
50X = [XFFT1(:,P/2+1:P) XFFT1(:,1:P/2)];
51
52% Downshift in frequency
53E = complex(zeros(Np,P));
54for idxl = -Np/2:Np/2-1    % length of l is Np
55    for k = 0:P-1          % length of k is P
56        E(idxl+Np/2+1,k+1) = exp(-1i*2*pi*k*L*idxl/Np);
57    end
58end
59XT = X.*E;                 % Output of Figure 3
60
61% Implement Figure 5 in (1): Implementation of the SSCA
62% Replication, repeat the k-th column of XT (XT(:,k)) L times
63XRep = complex(zeros(Np,P*L));
64for k = 1:P
65    XRep(:,(k-1)*L+1:k*L) = XT(:,k)*ones(1,L);
66end
67
68% Multiplication
69% Convert input vector into a matrix, the vector first becomes a row
70% vector, and repeat Np times. xc is (Np,N), the same size as XR
71inputRep = ones(Np,1)*input';
72XMul = conj((XRep.*inputRep)');
73
74% Centered N-point FFT (zero frequency in the 0 bin)
75XFFT2 = fftshift(fft(XMul));
76XFFT2 = [XFFT2(:,Np/2+1:Np) XFFT2(:,1:Np/2)]; % Output of Figure 5
77
78M = abs(XFFT2);
79
80% Convert M to Sx
81Sx = zeros(Np+1,2*N+1);
82for k1 = 1:N
83    k2 = 1:Np;
84    alpha = (k1-1)/N + (k2-1)/Np - 1;
85    f = ((k2-1)/Np - (k1-1)/N) / 2;
86    k = 1 + Np*(f+0.5);
87    n = 1 + N*(alpha+1);
88    for k2 = 1:Np
89        Sx(round(k(k2)),round(n(k2))) = M(k1,k2);
90    end
91end
