1function decisions = SpectrumSenseP25( ObjAGC, ObjSDRuReceiver, tx )
 2
 3            
 4% Collect Data
 5N = 10;
 6data = complex(zeros(tx.receiveBufferLength,N));
 7for k = 1:N
 8    tmp = step(ObjSDRuReceiver);
 9    data(:,k) = step(ObjAGC,tmp);
10end
11
12decisions = zeros(N,1);
13for k = 1:N
14    % 64 represents the window time Tw, 4*N represents the observation time T
15    %[Sx,alphao,fo] = commP25ssca(input,1,1/64,1/(4*N));
16    [Sx,alphao,fo] = commP25sscaGen(data(:,k),1,1/64,1/(tx.receiveBufferLength));
17    
18    [c,d] = size(Sx);
19    [Ades,Index] = sort(Sx(:),'descend');   % sort Sx by its element and store in Ades
20    [Ridx,Cidx]  = ind2sub(size(Sx),Index); % corresponding row index and column index
21    leng = length(Ades);
22    
23    decisions(k) = commP25decision_noise(Ades,Ridx,Cidx,leng,c,d);
24    
25end
26
27end
