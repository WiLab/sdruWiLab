1function [Response, previousMessage] = MACLayerRouter(...
  2    ObjAGC,...           %Objects
  3    ObjSDRuReceiver,...
  4    ObjSDRuTransmitter,...
  5    ObjDetect,...
  6    ObjPreambleDemod,...
  7    ObjDataDemod,...
  8    estimate,...         %Structs
  9    tx,...
 10    timeoutDuration,...  %Values/Vectors
 11    messageBits,...
 12    lookingForACK,...
 13    previousMessage...
 14    )
 15
 16% This function is called when the node is just listening to the spectrum
 17% waiting for a message to be transmitted to them
 18
 19%% Listen to the spectrum
 20% previousMessage will be updated for next run
 21[Response, previousMessage, originNodeID, destNodeID] = DLLayer(...
 22    ObjAGC,...           %Objects
 23    ObjSDRuReceiver,...
 24    ObjDetect,...
 25    ObjPreambleDemod,...
 26    ObjDataDemod,...
 27    estimate,...         %Structs
 28    tx,...
 29    timeoutDuration,...  %Values/Vectors
 30    messageBits,...
 31    previousMessage...
 32    );
 33
 34
 35
 36%% Possible response messages
 37% 1.) Timeout
 38% 2.) Some message
 39if ~strcmp(Response,'Timeout') &&  ~strcmp(Response,'Duplicate')
 40    %fprintf('###########################################\n');
 41    fprintf('MAC| Got message: %s\n',Response);
 42    fprintf('MAC| From Node: %d\n',int16(originNodeID));
 43    
 44    if ~lookingForACK
 45        fprintf('MAC| Transmitting ACK\n');
 46        % Send ACK
 47        fprintf('Transmitting to node: %d, with offset: %f\n',int16(originNodeID),tx.offsetTable(originNodeID));
 48        ObjSDRuTransmitter.CenterFrequency = tx.CenterFrequency + tx.offsetTable(originNodeID);% Adjust offset for node       
 49        PHYTransmit(...
 50            ObjSDRuTransmitter,...
 51            ObjSDRuReceiver,...
 52            'ACK',...
 53            tx.samplingFreq,...
 54            originNodeID,...
 55            destNodeID...
 56            );
 57        
 58    end
 59elseif strcmp(Response,'Duplicate')
 60    
 61    fprintf('MAC| Transmitting ACK for dupe\n');
 62    %% Send ACK
 63    fprintf('Transmitting to node: %d, with offset: %f\n',int16(originNodeID),tx.offsetTable(originNodeID));
 64    ObjSDRuTransmitter.CenterFrequency = tx.CenterFrequency + tx.offsetTable(originNodeID); % Adjust offset for node
 65    PHYTransmit(...
 66        ObjSDRuTransmitter,...
 67        ObjSDRuReceiver,...
 68        'ACK',...
 69        tx.samplingFreq,...
 70        originNodeID,...
 71        destNodeID...
 72        );
 73end
 74
 75% Send message to next user if required
 76if destNodeID ~= tx.nodeNum
 77
 78    recipient = destNodeID;
 79    %message = Response;
 80    
 81    % Send message to next node
 82    [previousMessage, msgStatus] = MACLayerTransmitter(...
 83        ObjAGC,...           %Objects
 84        ObjSDRuReceiver,...
 85        ObjSDRuTransmitter,...
 86        ObjDetect,...
 87        ObjPreambleDemod,...
 88        ObjDataDemod,...
 89        estimate,...         %Structs
 90        tx,...
 91        timeoutDuration,...  %Values/Vectors
 92        messageBits,...
 93        Response,...
 94        previousMessage,...
 95        recipient...
 96        );
 97    
 98    % What happend?
 99    if msgStatus
100        fprintf('MAC| Message redirect successful\n');
101    else
102        fprintf('MAC| Message redirect failed\n');
103    end
104else
105    fprintf('MAC| That one was for me:)\n');
106end
107
108
109end
