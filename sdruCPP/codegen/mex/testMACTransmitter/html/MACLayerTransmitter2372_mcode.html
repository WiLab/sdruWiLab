1function [previousMessage, msgStatus] = MACLayerTransmitter(...
 2    ObjAGC,...           %Objects
 3    ObjSDRuReceiver,...
 4    ObjSDRuTransmitter,...
 5    ObjDetect,...
 6    ObjPreambleDemod,...
 7    ObjDataDemod,...

18
19% % Sense spectrum and wait until it is unoccupied
20for tries = 1:4 % try only so many times

25%     if occupied

46
47

51    PHYTransmit(...
52        ObjSDRuTransmitter,...
53        ObjSDRuReceiver,...
54        message,...
55        tx.samplingFreq,...
56        tx.nodeNum,...%originator

65        ObjAGC,...           %Objects
66        ObjSDRuReceiver,...

74        messageBits,...
75        lookingForACK,...
76        previousMessage...
77        );
78
79    if strcmp(Response,'ACK')
80        fprintf('MAC| Got ACK\n');
81        msgStatus = true;
82        break
83    else
84        fprintf('MAC| Retransmitting message\n');
85    end
86    if tries >= 4
87        fprintf('###########################################\n');
